Quick Start

# Install Orca
cargo install --path src/crates/orca

# Initialize
orca init

# Configure LLM provider in ~/.orca/orca.toml
# [llm]
# provider = "anthropic"
# model = "claude-3-sonnet"
# api_key = "${ANTHROPIC_API_KEY}"

# Create and run a task
orca task create "Analyze project structure"
orca task run <task-id>
