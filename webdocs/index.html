Getting Started

Installation

Add acolib crates to your Cargo.toml:

[dependencies]
llm = { path = "src/crates/llm" }
tooling = { path = "src/crates/tooling" }
langgraph-core = { path = "src/crates/langgraph-core" }

Quick Example

Create a simple LLM chat application:

use llm::remote::OpenAiClient;
use llm::config::RemoteLlmConfig;
use langgraph_core::llm::{ChatModel, ChatRequest};
use langgraph_core::Message;

#[tokio::main]
async fn main() -> Result<(), Box> {
    let config = RemoteLlmConfig::from_env(
        "OPENAI_API_KEY",
        "https://api.openai.com/v1",
        "gpt-4"
    )?;
    let client = OpenAiClient::new(config);

    let request = ChatRequest::new(vec![
        Message::human("What is Rust?")
    ]);

    let response = client.chat(request).await?;
    println!("{}", response.message.text().unwrap());

    Ok(())
}

Build Commands

# Build all workspace crates
cargo build --release

# Build specific crate
cargo build -p llm --release
cargo build -p tooling --release

# Run tests
cargo test -p llm
cargo test -p tooling
