Configuration Management

Consistent patterns for loading and managing configuration from environment variables and config files.

Basic Usage

use tooling::config::{ConfigBuilder, get_env_parse, get_env_bool};

#[derive(Clone, Default)]
struct AppConfig {
    pub port: u16,
    pub debug: bool,
}

impl ConfigBuilder for AppConfig {
    fn from_env(prefix: &str) -> tooling::Result {
        use tooling::config::{get_env_parse_or, get_env_bool};

        Ok(Self {
            port: get_env_parse_or(&format!("{}PORT", prefix), 8080)?,
            debug: get_env_bool(&format!("{}DEBUG", prefix))?.unwrap_or(false),
        })
    }
}

// Load config from environment with defaults
let config = AppConfig::from_env_with_defaults("APP_")?;

Features

Function Description
get_env_parse Parse environment variable to any type
get_env_parse_or Parse with default value
get_env_bool Parse boolean environment variables
ConfigBuilder Trait for implementing config builders

Error Handling

Add context to errors and format error chains for better debugging.

Adding Context

use tooling::error::{ErrorContext, format_error_chain};

fn read_config() -> Result> {
    let contents = std::fs::read_to_string("config.json")
        .context("Failed to read configuration file")?;
    Ok(contents)
}

// Format error chains for display
match operation() {
    Err(e) => eprintln!("{}", format_error_chain(&*e)),
    Ok(_) => {}
}

Error Chain Formatting

// Output example:
// Error: Failed to read configuration file
// Caused by: No such file or directory (os error 2)

Async Utilities

Retry policies with exponential backoff and timeout utilities for async operations.

Retry with Exponential Backoff

use tooling::async_utils::retry::{RetryPolicy, with_retry};
use std::time::Duration;

// Configure retry policy
let policy = RetryPolicy::new(3)
    .with_initial_interval(1.0)
    .with_backoff_factor(2.0);

// Retry operation on failure
let result = with_retry(&policy, || async {
    call_external_api().await
}).await?;

Timeouts

use tooling::async_utils::timeout::{with_timeout, TimeoutGuard};
use std::time::Duration;

// Simple timeout
let result = with_timeout(
    Duration::from_secs(30),
    slow_operation()
).await?;

// RAII timeout guard
let guard = TimeoutGuard::new(Duration::from_secs(60));
guard.execute(async {
    // Operation with timeout
    Ok("done")
}).await?;

Exponential Backoff: Retry intervals increase exponentially (1s, 2s, 4s, 8s...) with optional jitter to prevent thundering herd problems.

Validation

Fluent validation API with chainable rules for validating data.

Number Validation

use tooling::validation::Validator;

let age = 25;
Validator::new(age, "age")
    .min(0)
    .max(120)
    .validate()?;

String Validation

let email = "user@example.com";
Validator::new(email, "email")
    .not_empty()
    .min_length(3)
    .max_length(100)
    .matches(r"^[^@]+@[^@]+\.[^@]+$")
    .validate()?;

Custom Validation

Validator::new(value, "value")
    .custom(|v| {
        if v % 2 == 0 {
            Ok(())
        } else {
            Err("Value must be even".to_string())
        }
    })
    .validate()?;

Collect All Errors

match Validator::new(value, "value")
    .min(0)
    .max(100)
    .validate_all() {
    Ok(v) => println!("Valid: {}", v),
    Err(errors) => {
        for error in errors {
            eprintln!("Error: {}", error);
        }
    }
}

Serialization

Stable JSON serialization and hashing utilities for cache keys and consistent output.

Stable JSON Serialization

use tooling::serialization::{
    generate_hash, generate_json_hash, stable_json_string,
    truncate_json, pretty_json
};
use serde_json::json;

// Stable serialization with sorted keys
let val = json!({"b": 2, "a": 1});
let stable = stable_json_string(&val)?;
// Output: {"a":1,"b":2}

Generate Cache Keys

// Generate hash from JSON for cache keys
let hash = generate_json_hash(&json_value);
println!("Cache key: {}", hash);

Truncate for Logging

let large_json = "...very long JSON string...";
let truncated = truncate_json(large_json, 100);
println!("Response: {}", truncated);

Rate Limiting

Token bucket and sliding window rate limiters for controlling request rates.

Token Bucket Rate Limiter

use tooling::rate_limit::RateLimiter;
use std::time::Duration;

// 100 requests per minute
let limiter = RateLimiter::new(100, Duration::from_secs(60));

if limiter.check().await {
    // Perform operation
} else {
    // Rate limited - deny request
}

// Wait until allowed (blocking)
limiter.acquire().await;
// Now perform operation

Sliding Window Limiter

use tooling::rate_limit::SlidingWindowLimiter;

// 1000 requests per minute with sliding window
let limiter = SlidingWindowLimiter::new(1000, Duration::from_secs(60));

if limiter.check().await {
    // Operation allowed
} else {
    // Too many requests
}

Token Bucket vs Sliding Window: Token bucket allows bursts up to the limit, while sliding window provides more even distribution over time.

Logging

Structured logging helpers and formatters for better observability.

RAII Logging Guard

use tooling::logging::LogGuard;

fn process() {
    let _guard = LogGuard::new("process");
    // Logs entry and exit automatically
    // "Entering process"
    // ... function body ...
    // "Exiting process"
}

Time Async Operations

use tooling::logging::timed;

let result = timed("api_call", async {
    call_api().await
}).await;
// Logs: "api_call completed in 1.23s"

Format Helpers

use tooling::logging::{format_duration, format_bytes, sanitize_for_logging};
use std::time::Duration;

let duration_str = format_duration(Duration::from_millis(1500));
// Output: "1.50s"

let size_str = format_bytes(1024 * 1024);
// Output: "1.00 MB"

let log = "API key: sk-abc123";
let safe = sanitize_for_logging(&log);
// Output: "API key: [REDACTED]"

Design Principles

1. Ergonomic APIs

Fluent, chainable interfaces that are easy to use and understand.

2. Type Safety

Leverage Rust's type system to catch errors at compile time.

3. Composability

Mix and match utilities to build exactly what you need.

4. Performance

Zero-cost abstractions where possible, efficient implementations.

5. Testing

Comprehensive test coverage for all modules and features.

6. Documentation

Clear examples and use cases for every feature.
